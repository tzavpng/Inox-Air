function getLang(){ return localStorage.getItem("lang") || "el"; }
function setLang(lang){ localStorage.setItem("lang", lang); }

function applyI18n(){
  const lang = getLang();
  const dict = I18N[lang] || I18N.el;

  document.documentElement.lang = lang;

  document.querySelectorAll("[data-i18n]").forEach(el => {
    const key = el.getAttribute("data-i18n");
    const val = dict[key];
    if (typeof val === "string") el.textContent = val;
  });

  document.querySelectorAll("[data-i18n-footer]").forEach(el => {
    const year = new Date().getFullYear();
    el.textContent = (dict.footer || "").replace("{year}", String(year));
  });

  const btn = document.getElementById("langToggle");
  if (btn) btn.textContent = dict.lang_btn || (lang === "el" ? "EN" : "EL");
}

function initLangToggle(){
  const btn = document.getElementById("langToggle");
  if (!btn) return;
  btn.addEventListener("click", () => {
    const next = getLang() === "el" ? "en" : "el";
    setLang(next);
    applyI18n();
  });
}

document.addEventListener("DOMContentLoaded", () => {
  applyI18n();
  initLangToggle();
});
